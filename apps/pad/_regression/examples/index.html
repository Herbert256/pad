Some random examples of PAD constructs, for a quick topview what PAD can do.

PHP
examples/hello.php 		HTML
examples/hello.html 		RESULT
<?php

  $hi = 'Hello World !';
  
?> 		<html>

  <body>
    <h1>{$hi}</h1>
  </body>

</html>
<?php

  $fruit = 'apple';
  
  $color = 'green';

?> 		<p>
  {if $fruit eq 'banana'}
    {if $color eq 'pink'}
      <b> A pink banana </b>
    {else}
      <b> Not a pink banana </b>
    {/if}
  {else}
    {if $color eq 'green'}
      <b> An {$fruit} is {$color}</b>
    {else}
      <b> Not a green fruit </b>
    {/if}
  {/if}
</p>
<?php

  $language = 'FR';

?> 		<p>
  {case $language}
    {when 'ES'}
      Hola Mundo
    {when 'NL'}
      Hallo Wereld
    {when 'FR'}
      Bonjour le Monde
    {when 'DE'}
      Hallo Welt
    {when 'IT'}
      Ciao Mondo
    {else}
      Hello World
  {/case}
</p>
{set $abc = 1}

{while $abc LE 10}
  <li>{$abc}
  {increment $abc}
{/while}
<?php 

  $myData = [
    [ "name" => "joe",   "phone" => "555-3425" ],
    [ "name" => "jim",   "phone" => "555-2364" ],
    [ "name" => "john",  "phone" => "555-7649" ],
    [ "name" => "jack",  "phone" => "555-3422" ],
    [ "name" => "jerry", "phone" => "555-4973" ]
  ];

?> 		<table border=1>
  {myData sort='phone'}
    <tr bgcolor="{switch 'pink', 'yellow'}">
      <td>{$name | ucwords}</td> 
      <td>{$phone}</td>
    </tr>
  {/myData}
</table>
