keys

Normally the keys of a PAD data collection are numbered 0..N,
but sometimes they have other values.

PHP
<?php

  $myData = [  
    'abc' => [ 'department' => 'IT',       'manager' => 'Bob'   ],
    'def' => [ 'department' => 'Kitchen',  'manager' => 'Jim'   ],
    'ghi' => [ 'department' => 'Helpdesk', 'manager' => 'Joe'   ],
    'jkl' => [ 'department' => 'Sales',    'manager' => 'Steve' ],
    'mno' => [ 'department' => 'Build',    'manager' => 'Jerry' ]
  ];

?>
HTML
<table border="1">
  {myData}
    {first}
      {keys} 
        <tr>
          <td>{$name}</td>
          {value}
            <td>
              {fields}
                {$name}: {$value} 
                <br>
              {/fields}
            </td>
          {/value}
        </tr>
      {/keys}
    {/first}
  {/myData}
</table>
